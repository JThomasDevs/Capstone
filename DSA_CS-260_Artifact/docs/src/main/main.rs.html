6/// Loads courses from a CSV file, validates prerequisites, and inserts them into the AVL tree.
7///
8/// # Arguments
9/// * `tree` - The AVLTree to insert courses into.
10/// * `filename` - The path to the CSV file containing course data.
11///
12/// Returns an io::Result indicating success or failure.
13pub fn load_courses_from_file(tree: &mut AVLTree, filename: &str) -> io::Result<()> {
14    // Open the file and read the lines into the 'lines' vector
15    let file = File::open(filename)?;
16    let reader = BufReader::new(file);
17    let mut lines: Vec<String> = Vec::new();
18
19    // trim whitespace from the lines and skip empty lines
20    for line in reader.lines() {
21        let line = line?;
22        if !line.trim().is_empty() {
23            lines.push(line);
24        }
25    }
26
27    // Gather all valid course IDs for later prerequisite validation
28    let valid_ids: HashSet<String> = lines
29        .iter()
30        .filter_map(|line| {
31            let id = line.split(',').next().unwrap_or("").trim();
32            if !id.is_empty() {
33                Some(id.to_string())
34            } else {
35                None
36            }
37        })
38        .collect();
39
40    // Iterate through each line in the 'lines' vector
41    for line in lines {
42        let split_line: Vec<&str> = line.split(',').collect();
43        // If the line has two elements, create a new Course with the first element as the Course ID and the second element as the Course name and no prerequisites
44        if split_line.len() == 2 {
45            let course = Course {
46                course_id: split_line[0].to_string(),
47                course_name: split_line[1].to_string(),
48                prereqs: Vec::new(),
49            };
50            tree.insert(course);
51        } 
52        // If the line has more than two elements, create a new Course with the first element as the Course ID and the second element as the Course name and the rest of the elements as prerequisites
53        else if split_line.len() > 2 {
54            let mut course = Course {
55                course_id: split_line[0].to_string(),
56                course_name: split_line[1].to_string(),
57                prereqs: Vec::new(),
58            };
59            for prereq in &split_line[2..] {
60                // If the prerequisite is valid, add it to the Course's prerequisites
61                if valid_ids.contains(*prereq) { // dereference the &&str to a &str
62                    course.prereqs.push(prereq.to_string());
63                }
64            }
65            // Insert the new Course into the AVL tree
66            tree.insert(course);
67        }
68    }
69    // Return Ok(()) to indicate success
70    Ok(())
71}

73/// Prints detailed information about a course if it exists, or an error if not.
74///
75/// # Arguments

77fn print_course(course: Option<&Course>) {
78    // Match against the Option enum to print the course information or an error message
79    match course {
80        // If the Course is Some, print the course information
81        Some(course) => {
82            println!("\nID: {}", course.course_id); // print the course ID
83            println!("Name: {}", course.course_name); // print the course name
84            if !course.prereqs.is_empty() { // if the course has prerequisites
85                print!("Prerequisites: "); // print the prerequisites
86                for (i, prereq) in course.prereqs.iter().enumerate() {
87                    if i != course.prereqs.len() - 1 {
88                        print!("{}, ", prereq);
89                    } else {
90                        println!("{}", prereq);
91                    }
92                }
93            }
94        }
95        // If the Course is None, print an error message
96        None => {
97            println!("\nERROR: Course not found.");
98        }
99    }

205/// AVL tree for storing courses, self-balancing on insertions and deletions.
206pub struct AVLTree {
207    /// Root node of the AVL tree.
208    root: Option<Box<Node>>,

211/// Enum to specify which child (left or right) to operate on.
212enum ChildSide {

228    /// Inserts a course into the AVL tree.
229    ///
230    /// # Arguments
231    /// * `course` - The Course to insert.
232    pub fn insert(&mut self, course: Course) {
233        // Create a new node with the Course, left and right children set to None, and height set to 0
234        let new_node = Box::new(Node {
235            course,
236            left: None,
237            right: None,
238            height: 0,

240        // Insert the new node into the AVL tree
241        self.root = Self::insert_node(self.root.take(), new_node);
242    }

244    /// Removes a node from the AVL tree by course ID, using the recursive approach from the root.
245    ///
246    /// # Arguments
247    /// * `course_id` - The ID of the course to remove.
248    pub fn remove(&mut self, course_id: &str) {
249        // Remove the node from the AVL tree
250        /* This method returns an Option<Box<Node>>, which represents the new tree root
251         * after the target node is removed. */
252        self.root = Self::remove_node(self.root.take(), course_id);

255    /// Searches for a node in the AVL tree by Course ID, starting at the root.
256    /// Returns a reference to the node if found, or None otherwise.
257    ///
258    /// # Arguments
259    /// * `course_id` - The ID of the course to search for.
260    ///
261    /// # Returns
262    /// * `Option<&Node>` - Some reference to the node if found, or None if not found.
263    pub fn search_node(&self, course_id: &str) -> Option<&Node> {
264        // Convert the Course ID to lowercase for case-insensitive comparison
265        let search_id = course_id.to_lowercase();
266        // Start at the root of the tree
267        let mut current = self.root.as_ref();
268        // Traverse the tree until the Course is found or the end of the tree is reached
269        while let Some(node) = current {
270            // Convert the Course ID to lowercase for case-insensitive comparison
271            let node_id = node.course.course_id.to_lowercase();
272            // If the search ID is less than the node ID, go left
273            if search_id < node_id {
274                current = node.left.as_ref();
275            } 
276            // If the search ID is greater than the node ID, go right
277            else if search_id > node_id {
278                current = node.right.as_ref();
279            } 
280            // If the search ID is equal to the node ID, return the node and end the loop
281            else {
282                return Some(node);
283            }
284        }
285        // If the Course is not found, return None
286        None
287    }

289    /// Performs a recursive in-order traversal of the AVL tree and prints courses.
290    pub fn in_order_traversal(&self) {
291        // Helper function to traverse the tree recursively
292        fn traverse(node: &Option<Box<Node>>) {
293            // If the node is None, return
294            let Some(cur) = node else { return };
295            // Traverse the left child, print the course, and traverse the right child
296            traverse(&cur.left); // keep going left until the left child is None
297            print_course(Some(&cur.course)); // print the course
298            traverse(&cur.right); // keep going right until the right child is None
299        }
300        // Start the traversal at the tree root
301        traverse(&self.root);
302    }

318    // --- Insertion Helper ---
319    /// Recursively inserts a node into the subtree, rebalancing as needed.
320    /// Returns the new subtree root after insertion and rebalancing.
321    ///
322    /// # Arguments
323    /// * `subtree` - The subtree to insert into.
324    /// * `node_to_insert` - The node to insert.
325    ///
326    /// # Returns
327    /// * `Option<Box<Node>>` - The new subtree root after insertion and rebalancing.
328    fn insert_node(subtree: Option<Box<Node>>, node_to_insert: Box<Node>) -> Option<Box<Node>> {
329        // Match against the subtree Node
330        match subtree {
331            // If the current_node is Some, compare the Course ID of the node to insert to the Course ID of the current node
332            Some(mut current_node) => {
333                // If the Course ID of the node to insert is less than the Course ID of the current node, go left
334                if node_to_insert.course.course_id < current_node.course.course_id {
335                    current_node.left = Self::insert_node(current_node.left.take(), node_to_insert);
336                } 
337                // If the Course ID of the node to insert is greater than the Course ID of the current node, go right
338                else if node_to_insert.course.course_id > current_node.course.course_id {
339                    current_node.right = Self::insert_node(current_node.right.take(), node_to_insert);
340                } 
341                // If the Course ID of the node to insert is equal to the Course ID of the current node, return the current node
342                else {
343                    return Some(current_node); // return the current node
344                }
345                // Rebalance the tree and return the new subtree root
346                Some(Self::rebalance(current_node))
347            }
348            // If the current_node is None, return the node to insert
349            None => {
350                Some(node_to_insert)
351            }
352        }
353    }

355    // --- Removal Helpers ---
356    /// Recursively removes a node by course ID and rebalances the tree.
357    /// Returns the new subtree root after removal and rebalancing.
358    ///
359    /// # Arguments
360    /// * `subtree` - The subtree to remove from.
361    /// * `course_id` - The ID of the course to remove.
362    ///
363    /// # Returns
364    /// * `Option<Box<Node>>` - The new subtree root after removal and rebalancing.
365    fn remove_node(subtree: Option<Box<Node>>, course_id: &str) -> Option<Box<Node>> {
366        // Match against the subtree Node
367        match subtree {
368            /* The passed subtree root node is Some */
369            // If the course ID is less than the node's course ID, go left
370            Some(mut node) if course_id < &node.course.course_id => {
371                // Recursively call remove_node on the left child
372                node.left = Self::remove_node(node.left.take(), course_id);
373                // Rebalance the tree and return the new subtree root
374                Some(Self::rebalance(node))
375            }
376            // If the course ID is greater than the node's course ID, go right
377            Some(mut node) if course_id > &node.course.course_id => {
378                // Recursively call remove_node on the right child
379                node.right = Self::remove_node(node.right.take(), course_id);
380                // Rebalance the tree and return the new subtree root
381                Some(Self::rebalance(node))
382            }
383            // If the course ID is equal to the node's course ID, remove the node
384            Some(mut node) => 
385            // Determine the successor node to replace the removed node based on the number of children
386            match (node.left.take(), node.right.take()) {
387                // If the node has no children, return None
388                (None, None) => None,
389                // If the node has only a left child, return the left child
390                (Some(left), None) => Some(left),
391                // If the node has only a right child, return the right child
392                (None, Some(right)) => Some(right),
393                // If the node has two children, find the in-order successor and return the new subtree root
394                (Some(_), Some(right)) => {
395                    // Find the in-order successor
396                    let (successor, new_right) = Self::find_successor(right);
397                    // Set the left child of the successor to the left child of the node
398                    let mut successor = successor;
399                    // Set the right child of the successor to the right child of the node
400                    successor.right = new_right;
401                    // Rebalance the tree and return the new subtree root
402                    Some(Self::rebalance(successor))
403                }
404            },
405            // If the subtree root node is None, return None
406            None => None,
407        }
408    }

410    /// Helper to find the in-order successor node to a node with two children
411    ///
412    /// # Arguments
413    /// * `node` - The right child subtree to search for the successor.
414    ///
415    /// # Returns
416    /// * `(Box<Node>, Option<Box<Node>>)` - The successor node and the new subtree root after removal.
417    fn find_successor(mut node: Box<Node>) -> (Box<Node>, Option<Box<Node>>) {
418        // Match against the left child of the node
419        match node.left.take() {
420            // If the left child is Some, recursively call find_successor on the left child
421            Some(left) => {
422                // Recursively call find_successor on the left child
423                let (successor, new_left) = Self::find_successor(left);
424                // Set the left child of the node to the new left child
425                node.left = new_left;
426                // Rebalance the tree and return the new subtree root
427                (successor, Some(Self::rebalance(node)))
428            }
429            // If the left child is None, return the node and its right child
430            None => {
431                // Set the right child of the node to the right child of the node
432                let right = node.right.take();
433                // Return the node and its right child
434                (node, right)
435            }
436        }
437    }

439    // --- AVL Balancing Helpers ---
440    /// Rebalances a node and returns the new subtree root.
441    /// This function updates the node's height, checks its balance factor, and performs
442    /// the appropriate single or double rotation to maintain AVL balance.
443    ///
444    /// # Arguments
445    /// * `node` - The node to rebalance.
446    ///
447    /// # Returns
448    /// * `Box<Node>` - The new subtree root after rebalancing.
449    fn rebalance(mut node: Box<Node>) -> Box<Node> {
450        // Update the height of the node
451        Self::update_height(&mut node);
452        // Get the balance factor of the node
453        let balance = Self::get_balance(&node);
454        // If the balance factor is -2, the node is right heavy
455        if balance == -2 {
456            // If the right child is Some and the balance factor of the right child is 1, perform a double rotation: Right-Left
457            if let Some(ref right_child) = node.right {
458                if Self::get_balance(right_child) == 1 {
459                    // If the right child is Some, perform a right rotation on the right child
460                    if let Some(right) = node.right.take() {
461                        // Set the right child of the node to the new right child
462                        node.right = Some(Self::rotate_right(right));
463                    }
464                }
465            }
466            return Self::rotate_left(node);
467        } else if balance == 2 {
468            // If the balance factor is 2, the node is left heavy
469            if let Some(ref left_child) = node.left {
470                // If the balance factor of the left child is -1, perform a double rotation: Left-Right
471                if Self::get_balance(left_child) == -1 {
472                    // If the left child is Some, perform a left rotation on the left child
473                    if let Some(left) = node.left.take() {
474                        // Set the left child of the node to the new left child
475                        node.left = Some(Self::rotate_left(left));
476                    }
477                }
478            }
479            // Perform a right rotation on the node
480            return Self::rotate_right(node);
481        }
482        // Return the node
483        node
484    }

486    /// Returns the balance factor of a node (left height - right height) using match statements.
487    ///
488    /// # Arguments
489    /// * `node` - The node to get the balance factor of.
490    ///
491    /// # Returns
492    /// * `i32` - The balance factor of the node.
493    fn get_balance(node: &Node) -> i32 {
494        // Get the height of the left child
495        let left_height = Self::height(&node.left);
496        // Get the height of the right child
497        let right_height = Self::height(&node.right);
498        // Return the difference between the left and right heights
499        left_height - right_height
500    }

502    /// Updates the height of a node based on the heights of its children.
503    ///
504    /// # Arguments
505    /// * `node` - The node to update the height of.
506    fn update_height(node: &mut Box<Node>) {
507        // Get the height of the left child
508        let left_height = match node.left {
509            Some(ref left_child) => left_child.height,
510            None => -1,
511        };
512        // Get the height of the right child
513        let right_height = match node.right {
514            Some(ref right_child) => right_child.height,
515            None => -1,
516        };
517        // Set the height of the node to the maximum of the left and right heights plus 1
518        node.height = std::cmp::max(left_height, right_height) + 1;
519    }

521    // --- Rotation Helpers ---
522    /// Performs a left rotation on the given node and returns the new subtree root.
523    ///
524    /// # Arguments
525    /// * `root` - The node to perform a left rotation on.
526    ///
527    /// # Returns
528    /// * `Box<Node>` - The new subtree root after the left rotation.
529    fn rotate_left(mut root: Box<Node>) -> Box<Node> {
530        // Match against the right child of the root
531        match root.right.take() {
532            // If the right child is Some, perform a left rotation
533            Some(mut right_child) => {
534                // Save right_child's left subtree
535                let right_left_child = right_child.left.take(); // save the left subtree of the right child
536                // Set the root as the left child of the right child
537                Self::set_child(&mut right_child, ChildSide::Left, Some(root));
538                // Set the right left child as the right child of the new left child (which is the old root)
539                if let Some(ref mut left_child) = right_child.left { // if the left child of the right child is Some
540                    Self::set_child(left_child, ChildSide::Right, right_left_child); // set the right left child as the right child of the left child
541                }
542                // Return the right child
543                right_child
544            },
545            // If the right child is None, return the original root
546            None => root,
547        }
548    }

550    /// Performs a right rotation on the given node and returns the new subtree root.
551    /// If the left child is None, returns the original root.
552    ///
553    /// # Arguments
554    /// * `root` - The node to perform a right rotation on.
555    ///
556    /// # Returns
557    /// * `Box<Node>` - The new subtree root after the right rotation.
558    fn rotate_right(mut root: Box<Node>) -> Box<Node> {
559        // Match against the left child of the root
560        match root.left.take() {
561            // If the left child is Some, perform a right rotation
562            Some(mut left_child) => {
563                // Save left_child's right subtree
564                let left_right_child = left_child.right.take(); // save the right subtree of the left child
565                // Set the root as the right child of the left child
566                Self::set_child(&mut left_child, ChildSide::Right, Some(root));
567                // Set the left right child as the left child of the new right child (which is the old root)
568                if let Some(ref mut right_child) = left_child.right { // if the right child of the left child is Some
569                    Self::set_child(right_child, ChildSide::Left, left_right_child); // set the left right child as the left child of the right child
570                }
571                // Return the left child
572                left_child
573            },
574            // If the left child is None, return the original root
575            None => root,
576        }
577    }

581    /// Updates the parent's height after setting the child.

583    /// # Arguments
584    /// * `parent` - The parent node to set the child of.

589        match side {
590            // If the side is Left, set the left child of the parent to the child

592            // If the side is Right, set the right child of the parent to the child
593            ChildSide::Right => parent.right = child,
594        }
595        // Update the height of the parent
596        Self::update_height(parent);
597    }
